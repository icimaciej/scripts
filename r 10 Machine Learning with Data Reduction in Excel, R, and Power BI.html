# determining k
maxK <- 10
k <- c(1:maxK)
SSE <- c(1:maxK)

for (i in k) {
  set.seed(100)
  results <- kmeans(averages[,c('Temperature','Rainfall')], i, nstart=100)
  SSE[i] <- results$tot.withinss
}

reduction <- c(1:maxK)
for (i in k) {
  if (i==maxK) {
    reduction[i] <- 0
  } else {
    reduction[i] <- SSE[i+1] - SSE[i]
  }
}
elbow <- data.frame(k, SSE, reduction)
# View(elbow)

plot(reduction ~ k, data=elbow, type="l")
